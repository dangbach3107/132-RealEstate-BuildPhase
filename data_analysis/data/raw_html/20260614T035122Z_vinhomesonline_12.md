function propertyDetail({ gallery, floorPlan, mortgageBase }) {
    return {
        gallery,
        floorPlan,
        activeImg: 0,
        lightboxOpen: false,
        lightboxFloor: false,
        tab: 'info',

        // Mortgage calculator state
        calc: {
            propertyPrice: mortgageBase > 0 ? +(mortgageBase / 1e9).toFixed(2) : 3,
            downPct: 30,
            rate: 8.5,
            term: 20,
            loanAmount: 0,
            downPayment: 0,
            monthly: 0,
            totalInterest: 0,
        },

        // AI narration state (same scope — direct access to calc)
        aiLoading: false,
        aiNarrationText: '',
        aiNarrationError: '',
        _aiPollTimer: null,

        init() {
            this.calcMortgage();
        },

        prevImg() {
            this.activeImg = (this.activeImg - 1 + this.gallery.length) % this.gallery.length;
        },
        nextImg() {
            this.activeImg = (this.activeImg + 1) % this.gallery.length;
        },

        calcMortgage() {
            const p = parseFloat(this.calc.propertyPrice) || 0;
            const down = p * (this.calc.downPct / 100);
            const loan = p - down;
            const monthlyRate = (parseFloat(this.calc.rate) || 0) / 100 / 12;
            const n = (parseInt(this.calc.term) || 20) * 12;
            let monthly = 0;
            if (monthlyRate > 0 && n > 0 && loan > 0) {
                monthly = loan * (monthlyRate * Math.pow(1 + monthlyRate, n)) / (Math.pow(1 + monthlyRate, n) - 1);
            }
            this.calc.loanAmount = loan;
            this.calc.downPayment = down;
            this.calc.monthly = monthly;
            this.calc.totalInterest = monthly * n - loan;
        },

        formatBillion(val) {
            if (!val || val <= 0) return '0';
            return parseFloat(val.toFixed(2)).toString();
        },
        formatMillionVND(valBillion) {
            const million = valBillion * 1000;
            return million >= 1000
                ? parseFloat((valBillion).toFixed(2)) + ' tỷ'
                : parseFloat(million.toFixed(1)).toString();
        },

        // AI narration — direct access to this.calc (same component scope)
        async requestNarration() {
            this.aiLoading = true;
            this.aiNarrationError = '';
            this.aiNarrationText  = '';
            clearInterval(this._aiPollTimer);

            const payload = {
                property_price:       parseFloat(this.calc.propertyPrice) * 1e9,
                down_payment_percent: this.calc.downPct,
                annual_rate:          this.calc.rate,
                term_months:          this.calc.term * 12,
            };

            try {
                const res = await fetch('/mapi/ai/narrate/mortgage', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Accept': 'application/json',
                        'X-CSRF-TOKEN': document.querySelector('meta[name=csrf-token]').content,
                    },
                    body: JSON.stringify(payload),
                });

                if (!res.ok) {
                    const j = await res.json().catch(() => ({}));
                    this.aiNarrationError = j.message ?? 'Lỗi yêu cầu diễn giải.';
                    this.aiLoading = false;
                    return;
                }

                const data = await res.json();

                if (data.status === 'cached' && data.narration_key) {
                    await this.pollNarration(data.narration_key);
                } else if (data.narration_key) {
                    this._aiPollTimer = setInterval(() => this.pollNarration(data.narration_key), 2000);
                }
            } catch {
                this.aiNarrationError = 'Lỗi kết nối.';
                this.aiLoading = false;
            }
        },

        async pollNarration(key) {
            try {
                const res = await fetch('/mapi/ai/narration/' + encodeURIComponent(key), {
                    headers: { 'Accept': 'application/json' },
                });
                if (!res.ok) return;
                const data = await res.json();
                if (data.ready) {
                    clearInterval(this._aiPollTimer);
                    this.aiLoading = false;
                    if (data.error) { this.aiNarrationError = 'AI không phản hồi. Thử lại sau.'; return; }
                    this.aiNarrationText = data.text ?? '';
                }
            } catch { /* retry on next tick */ }
        },
    };
}

function poiPanel(lat, lng) {
    return {
        lat,
        lng,
        pois: [],
        activeCategory: '',
        loadingPoi: false,
        poiError: '',

        async loadPoi(category) {
            if (this.activeCategory === category) return;
            this.activeCategory = category;
            this.pois = [];
            this.poiError = '';
            this.loadingPoi = true;

            try {
                const params = new URLSearchParams({ lat: this.lat, lng: this.lng, category });
                const res = await fetch('/mapi/gis/poi?' + params.toString(), {
                    headers: { 'Accept': 'application/json' },
                });
                if (!res.ok) { this.poiError = 'Không thể tải dữ liệu tiện ích.'; return; }
                const data = await res.json();
                this.pois = data.data ?? [];
            } catch {
                this.poiError = 'Lỗi kết nối.';
            } finally {
                this.loadingPoi = false;
            }
        },
    };
}
